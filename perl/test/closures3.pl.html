#!/usr/bin/perl -w

sub accum {
    my $curr = 0;
    sub {
        my $param = shift;
        $curr += $param;
    }
}


my $acc1 = accum();

print "a1 ",$acc1->(3),"\n";
print "a1 ",$acc1->(5),"\n";
